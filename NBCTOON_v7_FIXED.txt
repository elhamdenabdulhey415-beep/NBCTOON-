ALLOWED_IMAGE_TYPES = {\"image/jpeg\", \"image/png\", \"image/webp\", \"image/gif\"}

@api.post(\"/upload\")
async def upload_file(
    file: UploadFile = File(...),
    kind: str = Form(\"video\"),  # \"video\" or \"thumbnail\"
    user=Depends(get_admin_user),
):
    content_type = file.content_type or \"\"
    if kind == \"video\" and content_type not in ALLOWED_VIDEO_TYPES:
        raise HTTPException(400, f\"Invalid video type: {content_type}. Allowed: mp4, mkv, webm, mov, avi\")
    if kind == \"thumbnail\" and content_type not in ALLOWED_IMAGE_TYPES:
        raise HTTPException(400, f\"Invalid image type: {content_type}. Allowed: jpg, png, webp, gif\")
    target_dir = VIDEOS_DIR if kind == \"video\" else THUMBS_DIR
    ext = Path(file.filename or \"\").suffix or (\".mp4\" if kind == \"video\" else \".jpg\")
    fname = f\"{uuid.uuid4().hex}{ext}\"
    fpath = target_dir / fname
    total = 0
    max_size = MAX_VIDEO_SIZE if kind == \"video\" else MAX_IMAGE_SIZE
    async with aiofiles.open(fpath, \"wb\") as out:
        while True:
            chunk = await file.read(1024 * 1024)
            if not chunk:
                break
            total += len(chunk)
            if total > max_size:
                fpath.unlink(missing_ok=True)
                raise HTTPException(413, \"File too large\")
            await out.write(chunk)
    rel = f\"/api/uploads/{'videos' if kind == 'video' else 'thumbnails'}/{fname}\"
    return {\"url\": rel, \"filename\": fname, \"size\": total}
